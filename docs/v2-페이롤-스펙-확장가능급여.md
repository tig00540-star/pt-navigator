## 0. 설계 원칙 (합의)

- **"자동계산은 보조, 최종급여는 원장이 확정하는 편집 필드(=source of truth)."** 아는 구조면 자동 제안, 원장은 확인·수정 또는 무시하고 직접 입력.
- **모델을 하드코딩하지 않는다.** `pay_scheme.type` enum + `manual` 안전망 → 어떤 센터 구조든 (규칙이면 타입 추가 / 아니면 manual)로 항상 수용.
- **v1 자동타입 = `banded`(일반화) + `manual` 둘.** `banded` 하나로 매출구간%·수업수밴드×회당정액·단순회당고정 다 흡수. 나머지는 manual.
- **적용 단위 = 계정 기본 + 트레이너별 override.**
- **다음 증분(이번 X):** `time_slot`(수업 시각별 단가·자동), manual 간이 계산 보조.
- **회귀 안전:** 기존 `pay_policy`는 **건드리지 않고 남긴다**(마이그레이션이 pay_scheme로 seed만). 새 코드는 pay_scheme를 읽음.

## 1. 데이터 모델 (마이그레이션 2개 · Supabase 실행 + docs/migrations 기록)

### 1-1. `pay_scheme` — 급여 방식 설정 (스코프당 1행)

```sql
-- docs/migrations/2026-07-12-pay-scheme.sql
create table if not exists pay_scheme (
  id          uuid primary key default gen_random_uuid(),
  account_id  uuid not null,
  trainer_id  uuid null,                 -- null=계정 기본 / set=그 트레이너 개별 override
  type        text not null default 'banded',   -- 'banded' | 'manual'
  band_basis  text null,                 -- 'revenue' | 'session_count'  (type='banded'일 때만)
  bands       jsonb not null default '[]'::jsonb,
  -- bands 원소: { "min": 100, "payout_type": "flat_per_session"|"pct_of_price"|"fixed",
  --              "payout_value": 20000, "incentive_type": "pct"|"flat"|"none", "incentive_value": 0 }
  updated_at  timestamptz not null default now()
);

-- 스코프당 1행 보장: 계정 기본(null)은 계정당 하나, 트레이너별은 (account,trainer)당 하나.
create unique index if not exists pay_scheme_default_ux
  on pay_scheme(account_id) where trainer_id is null;
create unique index if not exists pay_scheme_trainer_ux
  on pay_scheme(account_id, trainer_id) where trainer_id is not null;

alter table pay_scheme enable row level security;

-- SELECT: 같은 계정이면 읽기(트레이너가 자기 급여 근거 확인). 쓰기: 원장만.
create policy pay_scheme_sel on pay_scheme for select
  using (account_id = auth_account_id());
create policy pay_scheme_ins on pay_scheme for insert
  with check (account_id = auth_account_id() and auth_is_owner());
create policy pay_scheme_upd on pay_scheme for update
  using (account_id = auth_account_id() and auth_is_owner())
  with check (account_id = auth_account_id() and auth_is_owner());
create policy pay_scheme_del on pay_scheme for delete
  using (account_id = auth_account_id() and auth_is_owner());
```

**기존 pay_policy → pay_scheme seed** (같은 마이그레이션 하단 · 있을 때만):
```sql
-- 기존 계정 밴드(매출기준·판매가%)를 계정 기본 banded 스킴으로 이관. pay_policy는 남겨둠(레거시).
insert into pay_scheme (account_id, trainer_id, type, band_basis, bands)
select p.account_id, null, 'banded', 'revenue',
       jsonb_agg(jsonb_build_object(
         'min', coalesce(p.min_amt,0),
         'payout_type', 'pct_of_price',
         'payout_value', coalesce(p.base_pct,0),
         'incentive_type', coalesce(p.incentive_type,'none'),
         'incentive_value', coalesce(p.incentive_value,0)
       ) order by p.min_amt)
from pay_policy p
group by p.account_id
on conflict do nothing;
```
> ⚠️ `account_id`가 pay_policy에 없으면 이 seed는 스킵하고, 원장이 UI에서 새로 세팅(빈 상태 가드). 클로드코드가 pay_policy 실제 컬럼 확인 후 seed 가능여부 판단.

## 2. lib — 계산 로직 (`lib/memberStatus.js` · 순수함수)

### 2-1. 스킴 해석 (트레이너별 override → 계정 기본 폴백)
```js
/** 트레이너별 개별 스킴 우선, 없으면 계정 기본(trainer_id=null), 그것도 없으면 null. */
export function resolveScheme(schemes, trainerId) {
  const arr = Array.isArray(schemes) ? schemes : [];
  return arr.find((s) => s.trainer_id === trainerId)
      ?? arr.find((s) => s.trainer_id == null)
      ?? null;
}
```

### 2-2. 수업수 집계 (session_count 기준 밴드용 · sessionPriceSumByTrainer 미러)
```js
/** ④ 밴드 기준 'session_count'용 — 이달 완료 수업 '개수'를 트레이너별로. sessionPriceSumByTrainer와 동일 필터. */
export function sessionCountByTrainer(logs, contracts, ym) {
  const trainerById = new Map();
  for (const c of (Array.isArray(contracts) ? contracts : [])) {
    if (c?.id) trainerById.set(c.id, c.trainer_id ?? "unknown");
  }
  const map = new Map();
  for (const l of (Array.isArray(logs) ? logs : [])) {
    if (!l || l.voided) continue;
    if (l.contract_id == null) continue;
    if (typeof l.session_at !== "string" || kstYm(l.session_at) !== ym) continue;
    const tid = trainerById.get(l.contract_id) ?? "unknown";
    map.set(tid, (map.get(tid) || 0) + 1);
  }
  return map;
}
```

### 2-3. 급여 디스패처 (기존 payForMonth 대체)
```js
/**
 * ④ 급여 자동제안 — scheme.type별 분기. manual이면 computed=null(원장 입력 대기).
 * @param scheme  resolveScheme 결과 (pay_scheme 1행 or null)
 * @param ctx     { monthRevenue, sessionCount, sessionPriceSum }
 * @returns { type, band, base, incentive, computed }  // computed=null이면 수동입력 필요
 */
export function payForScheme(scheme, { monthRevenue = 0, sessionCount = 0, sessionPriceSum = 0 } = {}) {
  if (!scheme || scheme.type === "manual") {
    return { type: "manual", band: null, base: 0, incentive: 0, computed: null };
  }
  // banded
  const bands = (Array.isArray(scheme.bands) ? scheme.bands : []).slice()
    .sort((a, b) => (a.min ?? 0) - (b.min ?? 0));
  const basisVal = scheme.band_basis === "session_count" ? sessionCount : monthRevenue;
  let band = null;
  for (const b of bands) { if (basisVal >= (b.min ?? 0)) band = b; }
  if (!band) return { type: "banded", band: null, base: 0, incentive: 0, computed: 0 };

  const pv = Number(band.payout_value ?? 0);
  let base = 0;
  if (band.payout_type === "pct_of_price") base = Math.round((sessionPriceSum * pv) / 100);
  else if (band.payout_type === "flat_per_session") base = sessionCount * pv;
  else if (band.payout_type === "fixed") base = pv;

  let incentive = 0;
  if (band.incentive_type === "pct") incentive = Math.round((monthRevenue * Number(band.incentive_value ?? 0)) / 100);
  else if (band.incentive_type === "flat") incentive = Number(band.incentive_value) || 0;

  return { type: "banded", band, base, incentive, computed: base + incentive };
}
```

> `payForMonth`(기존)는 **당장 지우지 말고 남겨둠** — admin·MyStats 콜사이트를 payForScheme로 옮긴 뒤(서브커밋 D·E), 콜사이트 0이 되면 다음 정리 커밋에서 제거.

## 3. admin — pay_scheme 편집 UI (`components/AdminPayrollSettings.jsx` 신설)

원장 전용. 계정 기본 스킴 + 트레이너별 override를 CRUD. admin/page.jsx 하단 "급여 정책" 섹션에 마운트.

**구조:**
- **탭/토글: 계정 기본 | 트레이너별.** (트레이너 목록은 admin이 이미 로드 중인 trainer 목록 재사용.)
- **스킴 에디터(공통 컴포넌트):**
  - `type` 라디오: **밴드 자동** / **수동(원장 직접 입력)**.
  - type=banded면: `band_basis` 라디오 **매출 기준 / 수업수 기준**.
  - `bands` 리스트(추가·삭제·정렬은 min 오름차순 자동): 각 밴드 =
    - **min** (매출기준이면 "○○원 이상", 수업수기준이면 "○개 이상")
    - **payout_type** 셀렉트: 판매가% / 회당정액 / 고정
    - **payout_value** (payout_type에 따라 라벨 동적: %/원·회당/원)
    - (선택) incentive_type(없음/판매가%/정액) + incentive_value
  - 저장 = pay_scheme upsert(스코프 1행). 데모가드(`if(!supabase)`)·`.select()` 하드닝·인라인 삭제확인 — **PtPricingSettings.jsx 패턴 그대로 재사용**.
- 트레이너별 탭에서 "계정 기본 따름"(override 없음=행 삭제) 옵션 제공.

**미리보기(권장):** 에디터 하단에 "예: 수업 160개·매출 500만원이면 급여 ○○원" 라이브 계산(payForScheme로) — 원장이 규칙 맞는지 즉시 확인. 환각 아니라 실제 함수 호출.

> 렌더 규칙: 색·클래스는 정적 토큰(text-ink/sub/primary 등)만, 동적 클래스 문자열 금지(CLAUDE.md). won()으로 금액 표기(이제 "200,000원").

## 4. admin — 급여 확정 흐름 (payroll_run 쓰기)

admin/page.jsx 트레이너 실적 카드의 "예상 급여" 자리를 **예상 → 확정** 흐름으로.

- 계산: `const scheme = resolveScheme(schemes, t.id); const pay = payForScheme(scheme, { monthRevenue: t.rev.total, sessionCount: sessCount.get(t.id)||0, sessionPriceSum: sessPriceSum.get(t.id)||0 });`
- 표시:
  - banded → "예상 급여 {won(pay.computed)}" + (밴드 근거 한 줄: 기준·payout 요약).
  - manual → "자동계산 없음 — 최종액 입력" (computed=null).
- **최종액 입력 + 확정:** 인풋(기본값 = 기존 payroll_run.final_total ?? pay.computed) + "확정" 버튼 → `payroll_run` upsert(account·trainer·ym, computed_total=pay.computed, final_total=입력값, scheme_snapshot=scheme, updated_at). 확정되면 "확정됨 {won(final)}" 배지 + 재편집 가능(lock은 선택: locked_at 있으면 읽기전용, 잠금해제 버튼).
- ym = 현재 조회 중인 달(admin의 기존 `ym`).
- payroll_run 로드: admin 초기 로드에 `supabase.from("payroll_run").select("*").eq("ym", ym)` 추가 → Map(trainer_id→run).

> 핵심: **final_total이 진실.** 표시·리포트(기능4)·트레이너 뷰 전부 `final_total ?? computed`를 쓴다.

## 7. 회귀·안전

- 기존 `pay_policy`·`payForMonth`는 **안 지운다**(콜사이트 다 옮긴 뒤 마지막에 정리). 전환 중에도 앱 안 깨짐.

- 새 테이블·RLS는 계정/원장 스코프라 트레이너가 남의 급여·정책 못 봄(payroll_run SELECT는 본인만).

## 8. 테스트 체크리스트

1. `npm.cmd run lint` green (각 서브커밋).
2. **마이그레이션:** Supabase에서 두 테이블 생성 + seed 확인. RLS로 다른 계정·다른 트레이너 접근 차단(원장 전체/트레이너 본인만).
3. **payForScheme 단위 검증(수기 or 콘솔):**
   - 매출밴드×판매가%: 기존 payForMonth와 같은 값 나오나(이관 정합).
   - 수업수밴드×회당정액: 수업 160개·밴드[100→2만,150→3만] → 480만원. 120개 → 240만원. 90개 → 최저밴드/0.
   - 고정: payout_type=fixed → payout_value 그대로.
   - manual → computed=null.
4. **override 우선순위:** 트레이너별 스킴 있으면 그것, 없으면 계정 기본. resolveScheme 확인.
5. **확정 흐름:** 예상 → 최종액 수정 → 확정 → payroll_run 저장 → 재방문 시 확정치 표시(final 우선). 원장만 쓰기.
6. **MyStats:** 트레이너가 자기 급여 확정치/예상 정상 표시, 편집 불가.
7. **회귀:** pay_scheme 없는 상태에서도 크래시 없이 "입력 대기"로 뜨는지. 기존 매출/집계 화면 무변.
8. 폰: Vercel 배포 후 하드리프레시 육안.

## 9. 후속 (다음 증분 — 이번 X)

- **`time_slot` 자동타입:** 수업 시각별 단가. `daily_workout_log.session_at`(KST 시)로 슬롯 매칭·합산. pay_scheme.type='time_slot' + slots jsonb([{start,end,rate}]) + payForScheme 분기 추가. 데이터 이미 있어 소규모.
- **manual 간이 계산 보조:** "회당단가 × 적용 수업수" 입력하면 앱이 곱해주는 중간 단계(완전 자동 아님).
- **pay_policy·payForMonth 제거:** 콜사이트 0 확인 후 정리 커밋.
- **급여명세 출력:** payroll_run 기반 월 명세(기능4 월간 리포트와 연계).
